```sql
notifications (
  id UUID, user_id UUID, subscription JSONB,  -- PushSubscription JSON

)
notification_thresholds (
  id UUID, user_id UUID, threshold_minutes INT, tone TEXT, -- 'gentle'|'urgent'|'final'
  created_at TIMESTAMPTZ
)
```
